计算R(n)的DTFT
w = -2*pi:(2*pi)/1000:2*pi;
X = (1-exp(-1i*w*length(x(x==1))))./(1-exp(-1i*w)); % R(n)的DTFT
magX = abs(X); % X的幅值
angX = angle(X);% X的相位
plot(w,magX);
xticks([-3*pi -2*pi -pi 0 pi 2*pi 3*pi])
xticklabels({'-3\pi','-2\pi','-\pi','0','\pi','2\pi','3\pi'})
grid on
plot(w,angX);
xticks([-3*pi -2*pi -pi 0 pi 2*pi 3*pi])
xticklabels({'-3\pi','-2\pi','-\pi','0','\pi','2\pi','3\pi'})
grid on
